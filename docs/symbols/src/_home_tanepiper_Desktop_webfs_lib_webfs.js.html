 62     /**
 63      * Creates a stats object
 64      * @private
 65      * @param   {FileEntry, DirectoryEntry}   entry
 66      * @param   {Function}                    callback
 67      * @type    {void}
 68      */
 69     var Stats = function(entry, callback) {
 70         var _self = {};
 71         
 72         if (entry.isFile) {
 73             
 74             entry.file(function(file) {
 75                 
 76                 var t = file.lastModifiedDate;
 77                 var m = t.getMonth() + 1;
 78                 var month = m < 10 ? "0" + m : m;
 79             
 80                 var time = [t.getFullYear(), month, t.getDate()].join('-') + 'T' + [t.getHours(), t.getMinutes(), t.getSeconds()].join(':') + 'Z';
 81             
 82                 _self.mtime = time;
 83                 _self.atime = time;
 84                 _self.ctime = time;
 85                 
 86                 _self.size = file.fileSize;
 87                 
 88                 afterMetaData()
 89             });
 90         } else {
 91             entry.getMetadata(function(metadata) {
 92                 var t = metadata.modificationTime;
 93                 var m = t.getMonth() + 1;
 94                 var month = m < 10 ? "0" + m : m;
 95                 
 96                 var time = [t.getFullYear(), month, t.getDate()].join('-') + 'T' + [t.getHours(), t.getMinutes(), t.getSeconds()].join(':') + 'Z';
 97                 
 98                 _self.mtime = time;
 99                 _self.atime = time;
100                 _self.ctime = time;
101                 
102                 _self.size = 0;
103                 
104                 afterMetaData()
105             });
106         }
107         
108         function afterMetaData() {
109             
110             _self.dev = 0;
111             _self.ino = 0;
112             _self.mode = 0;
113             _self.nlink = 0;
114             _self.uid = 0;
115             _self.gid = 0;
116             _self.rdev = 0;
117             _self.blocks = 0;
118             
119             
120             _self.isDirectory = entry.isDirectory;
121             _self.isFile = entry.isFile;
122             
123             /**
124              * These next stats functions all return false for nodejs compatibility
125              */
126             _self.isBlockDevice = false;
127             _self.isCharacterDevice = false;
128             _self.isSymbolicLink = false;
129             _self.isFIFO = false;
130             _self.isSocket = false;
131             
132             callback(null, _self);
133         }

144     /**
145      * Error handler for file system operations
146      * @param   {Error}     error
147      * @type    {void}
148      */
149     WebFS.prototype.errorHandler = function(error) {
150         var msg;
151         
152         switch(error.code) {
153             case this.NOT_FOUND_ERR:
154                 error.message = "The file or directory has not been found";
155                 break;
156             case this.SECURITY_ERR:
157                 error.message = "The file you are attempting to access is unsafe for web access or may be being accessed too many times.";
158                 break;
159             case this.ABORT_ERR:
160                 error.message = "The current operation has been aborted";
161                 break;
162             case this.NOT_READABLE_ERR:
163                 error.message = "The file you are attempting to read is not readable, this may be a permissions issue.";
164                 break;
165             case this.ENCODING_ERR:
166                 error.message = "The data or URL passed is malformed";
167                 break;
168             case this.NO_MODIFICATION_ALLOWED_ERR:
169                 error.message = "The file or directory cannot be modified.";
170                 break;
171             case this.INVALID_STATE_ERR:
172                 error.message = "The file or directory state has changed since the last operation.";
173                 break;
174             case this.SYNTAX_ERR:
175                 error.message = "There is a syntax error with this file operation.";
176                 break;
177             case this.INVALID_MODIFICATION_ERR:
178                 error.message = "Invalid file operation.";
179                 break;
180             case this.QUOTA_EXCEEDED_ERR:
181                 msg = "The quota for the filesystem has been exceeded.";
182                 break;
183             case this.TYPE_MISMATCH_ERR:
184                 error.message = "Incorrect file operation on file or directory.";
185                 break;
186             case this.PATH_EXISTS_ERR:
187                 error.message = "This path already exists";
188                 break;
189         }
190         
191         return error;
192     };

226     /**
227      * Rename or move src to dest.  If dest is a directory, must contain a trailing '/' char.
228      * @param   {String}    src 
229      * @param   {String}    dest
230      * @param   {Function}  callback
231      * @type    {void}
232      */
233     WebFS.prototype.rename = function(src, dest, callback) {
234         var _self = this;
235         
236         var errorHandler = function(error) {
237             callback(_self.errorHandler(error));
238         };
239         
240         var doMove = function(srcDirEntry, destDirEntry, newName) {
241             var name = newName || null;
242             srcDirEntry.moveTo(destDirEntry, name, function(newDirEntry) {
243                 callback(null, newDirEntry);
244             }, errorHandler);
245         };
246         
247         if (dest[dest.length - 1] == _self.DIR_SEPARATOR) {
248             _self.root.getDirectory(src, {}, function(srcDirEntry) {
249                 // Create blacklist for dirs we can't re-create.
250                 var create = _self.DIR_BLACKLIST.indexOf(dest) != -1 ? false : true;
251              
252                 _self.root.getDirectory(dest, {create: create}, function(destDirEntry) {
253                     doMove(srcDirEntry, destDirEntry);
254                 }, errorHandler);
255              }, function(error) {
256                  // Try the src entry as a file instead.
257                 _self.root.getFile(src, {}, function(srcDirEntry) {
258                     _self.root.getDirectory(dest, {}, function(destDirEntry) {
259                         doMove(srcDirEntry, destDirEntry);
260                     }, errorHandler);
261                 }, errorHandler);
262             });
263         } else {
264             // Treat src/destination as files.
265             _self.root.getFile(src, {}, function(srcFileEntry) {
266                 srcFileEntry.getParent(function(parentDirEntry) {
267                     doMove(srcFileEntry, parentDirEntry, dest);
268               }, errorHandler);
269             }, errorHandler);
270         }
271     };

431     /**
432      * Creates a directory on the filesystem, will recursivly create paths
433      * @param   {String}    path
434      * @param   {Number}    mode
435      * @param   {Function}  callback
436      * @type    {void}
437      */
438     WebFS.prototype.mkdir = function(path, mode, callback) {
439         var _self = this;
440         
441         if (typeof callback != "function") {
442             callback = mode;
443         }
444         
445         var errorHandler = function(error) {
446             callback(_self.errorHandler(error));
447         };
448         
449         var createDir = function(rootDir, folders) {
450             if (folders[0] == '.' || folders[0] == '') {
451                 folders = folders.slice(1);
452             }
453 
454             rootDir.getDirectory(folders[0], {create: true}, function(dirEntry) {
455                 if (folders.length) {
456                     createDir(dirEntry, folders.slice(1));
457                 } else {
458                     callback(null, dirEntry);
459                 }
460             }, errorHandler);
461         };
462         createDir(this.fs.root, path.split('/'));
463     };

465     /**
466      * Reads the contents of a directory, returns the result as an array of entries
467      * @param   {String}    path
468      * @param   {Function}  callback
469      * @type    {void}
470      */
471     WebFS.prototype.readdir = function(path, callback) {
472         var _self = this;
473         
474         var errorHandler = function(error) {
475             callback(_self.errorHandler(error));
476         };
477         
478         var listHandler = function(dirHandler) {
479             var dirReader = dirHandler.createReader();
480             var entries = [];
481             var readEntries = function() {
482                 dirReader.readEntries(function(results) {
483                     if (!results.length) {
484                         callback(null, entries.sort());
485                     } else {
486                         for (var i = 0, j = results.length; i < j; i++) {
487                             if (results[i].isDirectory)
488                                 entries.push(results[i].fullPath + '/');
489                             else
490                                 entries.push(results[i].fullPath);
491                         }
492                         //entries = entries.concat(Array.prototype.slice.call(results || [], 0));
493                         readEntries();
494                     }
495                 }, errorHandler);
496             };
497             readEntries();
498         };
499         
500         _self.root.getDirectory(path, {}, listHandler, errorHandler);
501     };

514     /**
515      * Opens a file or directory and return a handler
516      * @param   {String}    path
517      * @param   {String}    flags
518      * @param   {Number}    mode
519      * @param   {Function}  callback
520      * @type    {void}
521      */
522     WebFS.prototype.open = function(path, flags, mode, callback) { 
523         var _self = this;
524         
525         if (typeof callback != "function") {
526             callback = mode;
527         }
528         if (typeof callback != "function") {
529             callback = flags;
530         }
531         
532         var errorHandler = function(error) {
533             if (error && error.code == _self.TYPE_MISMATCH_ERR) {
534                 _self.root.getDirectory(path, options, successHandler, function(error) {
535                     callback(_self.errorHandler(error));
536                 });
537             }
538         };
539         
540         var successHandler = function(fileHandler) {
541             callback(null, fileHandler);
542         };
543         
544         var options = {};
545         // If the flag is to write or append, and the file does not exist
546         // then we need to ensure it's created
547         if (['w', 'w+', 'a', 'a+'].indexOf(flags) > -1)
548             options.create = true;
549 
550         _self.root.getFile(path, options, successHandler, errorHandler);
551     };

553     /**
554      * Writes the contents of a Blob or File to a FileEntry on the filesystem
555      * @param   {FileEntry} fileHandler
556      * @param   {Mixed}     buffer
557      * @param   {Number}    offset
558      * @param   {Number}    length
559      * @param   {Number}    position
560      * @param   {Function}  callback
561      * @type    {void}
562      */
563     WebFS.prototype.write = function(fileHandler, buffer, offset, length, position, callback) {
564         var _self = this,
565             data;
566             
567         if (typeof callback != "function") {
568             callback = position;
569         }
570         if (typeof callback != "function") {
571             callback = length;
572         }
573         if (typeof callback != "function") {
574             callback = offset;
575         }
576             
577         var errorHandler = function(error) {
578             callback(_self.errorHandler(error));
579         };
580                 
581         var data = (typeof buffer == 'string') ? _self.createBlob(buffer) : buffer;
582         
583         var writerHandler = function(fileWriter) {
584             
585             fileWriter.onwriteend = function(e) {
586                 callback(null, e.loaded, buffer);
587             };
588             
589             fileWriter.onerror = errorHandler;
590           
591             fileWriter.write(data);
592         };
593             
594         fileHandler.createWriter(writerHandler, errorHandler);
595     };

597     /**
598      * Asynchronously writes data to a file, replacing the file if it already exists.  Data can be a string or a buffer.
599      * @param   {String}    filename
600      * @param   {Mixed}     data
601      * @param   {String}    encoding
602      * @param   {Function}  callback
603      * @type    {void}
604      */
605     WebFS.prototype.writeFile = function(filename, data, encoding, callback) {
606         var _self = this;
607         
608         if (typeof callback != 'function') {
609             callback = encoding;
610         }
611         
612         var buffer = (typeof data == 'string') ? _self.createBlob(data) : data;
613         
614         var openFileHandler = function(error, fileHandler) {
615             _self.truncate(fileHandler, 0, function(error) {
616                 if (error)
617                     return callback(error);
618                 _self.write(fileHandler, buffer, null, null, null, function(error, written, buffer_) {
619                     callback(error, buffer_);  
620                 });
621             });
622         };
623         
624         _self.open(filename, 'w', null, openFileHandler);  
625     };

628      * Read data from the file specified by file handler.
629      * @param   {FileEntry} fileHandler
630      * @param   {Mixed}     buffer
631      * @param   {Number}    offset
632      * @param   {Number}    length
633      * @param   {Number}    position
634      * @type    {void}
635      */
636     WebFS.prototype.read = function(fileHandler, buffer, offset, length, position, callback) {
637         var _self = this,
638             data;
639             
640         if (typeof callback != "function") {
641             callback = position;
642         }
643         if (typeof callback != "function") {
644             callback = length;
645         }
646         if (typeof callback != "function") {
647             callback = offset;
648         }
649         
650         var errorHandler = function(error) {
651             callback(_self.errorHandler(error));
652         };
653         
654         fileHandler.file(function(file) {
655             var reader = new FileReader();
656             
657             reader.onloadend = function(e) {
658                 buffer.append(this.result);
659                 callback(null, buffer.getBlob().size, buffer.getBlob());
660             };
661             
662             reader.onerror = errorHandler;
663             
664             // Since we want to support binary or string data, we should read
665             // as an array buffer and allow the user to determine the output
666             // from the Blob/File interface buffer
667             reader.readAsArrayBuffer(file);
668         });
